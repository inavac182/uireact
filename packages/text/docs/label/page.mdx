import { UiLabel } from '@uireact/text';
import packageJson from './package-metadata.json';
import { Metadata } from '@uireact/docs-tools';

# UiLabel

<Metadata packageName='text' packageJson={packageJson} description='Component for rendering labels, commonly used in forms' includeInformation />

## Usage

```jsx live scope={{UiLabel}}
  <UiLabel>Some Label</UiLabel>
```

### UiLabel with color category

```jsx live scope={{UiLabel}}
<>
  <UiLabel category="error">Some label</UiLabel> <br />
  <UiLabel category="positive">Some label</UiLabel> <br />
  <UiLabel category="negative">Some label</UiLabel> <br />
</>
```

### UiLabel with default coloration

```jsx live scope={{UiLabel}}
<>
  <UiLabel coloration="light">Will aways render LIGHT coloration</UiLabel> <br />
  <UiLabel coloration="dark">Will aways render DARK coloration</UiLabel>
</>
```

### UiLabel with spacing

```jsx live scope={{UiLabel}}
<>
  <UiLabel padding={{all: 'five'}}>Will aways render LIGHT coloration</UiLabel>
</>
```
