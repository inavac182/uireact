import { UiText } from '@uireact/text';

import { UiList, UiListItem } from '@uireact/list';
import packageJson from './package-metadata.json';
import { Metadata } from '@uireact/docs-tools';

# UiList

<Metadata packageName='list' packageJson={packageJson} description='Component used to render lists, bulleted, ordered or unordered' includeInformation />

## UiList - default

```jsx live scope={{UiList, UiListItem, UiText}}
  <UiList>
    <UiListItem>
      <UiText>Item 1</UiText>
    </UiListItem>
    <UiListItem>
      <UiText>Item 2</UiText>
    </UiListItem>
  </UiList>
```

## UiList - ordered

```jsx live scope={{UiList, UiListItem, UiText}}
  <UiList type="ORDERED">
    <UiListItem>
      <UiText>Item 1</UiText>
    </UiListItem>
    <UiListItem>
      <UiText>Item 2</UiText>
    </UiListItem>
  </UiList>
```

## UiList - bulleted

```jsx live scope={{UiList, UiListItem, UiText}}
  <UiList type="BULLETED">
    <UiListItem>
      <UiText>Item 1</UiText>
    </UiListItem>
    <UiListItem>
      <UiText>Item 2</UiText>
    </UiListItem>
  </UiList>
```

## UiList - nested

```jsx live scope={{UiList, UiListItem, UiText}}
  <UiList type="BULLETED">
    <UiListItem>
      <UiText>Item 1</UiText>
      <UiList type="ORDERED">
        <UiListItem>
          <UiText>Item 1.1</UiText>
        </UiListItem>
        <UiListItem>
          <UiText>Item 1.2</UiText>
        </UiListItem>
      </UiList>
    </UiListItem>
    <UiListItem>
      <UiText>Item 2</UiText>
    </UiListItem>
  </UiList>
```
