import { UiSelect } from '@uireact/form';
import packageJson from './package-metadata.json';
import { Metadata } from '@uireact/docs-tools';

# UiSelect
<Metadata packageName='form' packageJson={packageJson} description='Component used to render select inputs' includeInformation />

## Usage

```jsx live scope={{UiSelect}}
  <UiSelect
    onChange={(value) => {
      console.log(value);
    }}
    defaultValue=""
  >
    <option disabled value="">
      Select a car
    </option>
    <option> Ford </option>
    <option> Chevy </option>
  </UiSelect>
```

## UiSelect with selected value

```jsx live scope={{UiSelect}}
  <UiSelect label="Car" labelOnTop value="Chevy">
    <option disabled>Select a car</option>
    <option>Ford</option>
    <option>Chevy</option>
  </UiSelect>
```

## UiSelect disabled

```jsx live scope={{UiSelect}}
  <UiSelect label="Car" disabled>
    <option> Ford </option>
    <option> Chevy </option>
  </UiSelect>
```

## UiSelect with error state and error message

```jsx live scope={{UiSelect}}
  <UiSelect label="Car" error="Some error" category="error">
    <option> Ford </option>
    <option> Chevy </option>
  </UiSelect>
```
