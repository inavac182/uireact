import { UiInput } from '@uireact/form';

import { UiSpacing } from '@uireact/foundation';

import { Metadata } from '@uireact/docs-tools';

# UiInput
<Metadata packageName='form' packageJson={packageJson} description='Component used to render input fields' includeInformation />

## UiInput with placeholder

```jsx live scope={{UiInput}}
  <UiInput
    placeholder="First Name"
    onChange={(value) => {
      console.log(value);
    }}
  />
```

## UiInput with size

```jsx live scope={{UiInput}}
  <UiInput
    placeholder="First Name"
    onChange={(value) => {
      console.log(value);
    }}
    size="xlarge"
  />
```

## UiInput with label

```jsx live scope={{UiInput}}
  <UiInput label="First Name" labelOnTop />
```

## UiInput with icon

```jsx live scope={{UiInput, UiIcon}}
  <UiInput label="Search" placeholder="Type your search" labelOnTop icon={<UiIcon icon="Search" />} />
```

## UiInput with override padding

```jsx live scope={{UiInput}}
  <UiInput label="First Name" padding={{ all: 'five' }} />
```

## UiInput disabled

```jsx live scope={{UiInput}}
  <UiInput label="First Name" disabled />
```

## UiInput with state

```jsx live scope={{UiInput}}
  <UiInput label="First Name" category="positive" />
```

## UiInput with error state and error message

```jsx live scope={{UiInput}}
  <UiInput label="First Name" error="Some error" category="error" />
```

## UiInput for passwords

```jsx live scope={{UiInput}}
  <UiInput label="Password" labelOnTop type="password" />
```

## UiInput for numbers

```jsx live scope={{UiInput}}
  <UiInput label="Number" labelOnTop type="number" value="123" />
```

## UiInput for datepicker

This is the browser's basic datepicker, each browser would render it the way it understand it. If you want a controlled datepicker look at the [datepicker input](./datepicker-input) component.

```jsx live scope={{UiInput}}
  <UiInput label="Date" labelOnTop type="date" />
```
